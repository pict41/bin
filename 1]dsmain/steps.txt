1.sudo apt install openjdk-17-jdk -y
2.java -version
3.sudo apt install ssh -y
4.ssh-keygen -t rsa -P ""
5.cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
6.chmod 600 ~/.ssh/authorized_keys
7.ssh localhost 
8.wget https://dlcdn.apache.org/hadoop/common/hadoop-3.5.0/hadoop-3.5.0.tar.gz
9.tar -xzf hadoop-3.5.0.tar.gz
10.sudo mv hadoop-3.5.0 /usr/local/hadoop

set Env in .bashrc

export JAVA_HOME=/usr/lib/jvm/java-17-openjdk-amd64
export HADOOP_HOME=/usr/local/hadoop
export PATH=$PATH:$HADOOP_HOME/bin
export PATH=$PATH:$HADOOP_HOME/sbin
export HADOOP_MAPRED_HOME=$HADOOP_HOME

export HADOOP_COMMON_HOME=$HADOOP_HOME
export HADOOP_HDFS_HOME=$HADOOP_HOME
export YARN_HOME=$HADOOP_HOME
export HADOOP_COMMON_LIB_NATIVE_DIR=$HADOOP_HOME/lib/native
export HADOOP_OPTS="-Djava.library.path=$HADOOP_HOME/lib"
export HADOOP_CONF_DIR=/usr/local/hadoop/etc/hadoop

source .bashrc

11.cd $HADOOP_HOME/etc/hadoop
12.nano core-site.xml
13.add 
	<property>
	<name>fs.defaultFS</name>
	<value>hdfs://localhost:9000</value>
	</property>
14.nano hdfs-site.xml
15.add
	<property>
	<name>dfs.replication</name>
	<value>1</value>
	</property>
	<property>
	<name>dfs.namenode.name.dir</name>
	<value>file:///usr/local/hadoop/hdfs/namenode</value>
	</property>
	<property>
	<name>dfs.datanode.data.dir</name>
	<value>file:///usr/local/hadoop/hdfs/datanode</value>
	</property>
	<property>
	<name>dfs.namenode.rpc-address</name>
	<value>localhost:9000</value>
	</property>
	<property>
	<name>dfs.namenode.http-address</name>
	<value>localhost:9870</value>
	</property>
16.nano mapred-site.xml
	<property>
	<name>mapreduce.framework.name</name>
	<value>yarn</value>
	</property>
17.nano yarn-site.xml
	<property>
	<name>yarn.resourcemanager.address</name>
	<value>localhost:8032</value>
	</property>
	<property>
	<name>yarn.nodemanager.aux-services</name>
	<value>mapreduce_shuffle</value>
	</property>
18.nano hadoop-env.sh
19.add
	 export JAVA_HOME=/usr/lib/jvm/java-17-openjdk-amd64
